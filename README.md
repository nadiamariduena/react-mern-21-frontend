### Now export the component inside the Home.jsx

```javascript
import React from "react";
//
//
//

  return (
    <div>
      <Announcement />
      <Navbar />
      <Slider />
      <Categories />
      <Products />
      <Newsletter />
    </div>
  );

export default Home;
```
